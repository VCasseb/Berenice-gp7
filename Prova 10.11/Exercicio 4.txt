#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void ler();


int main()
{
    FILE *fp;

    fp=fopen("Teste51.txt","w");
    char texto[100];
    char checagem[100]= {'Eng.Soft UA'};
    char frase[100]= {"O sonho da Berenice eh Utilizar alocacao dinamica"};
    if(fp==NULL)
    {
        printf("Erro na abertura do Arquivo.\n");
        return 1;
    }

    printf("Digite o texto:\n");
    gets(texto);
    if(strcmp(texto,checagem))
    {
        fprintf(fp,"%s",frase);
    }
    else
    {
        fprintf(fp,"%s", texto);
    }

    fclose(fp);
    ler();
    return 0;
}
